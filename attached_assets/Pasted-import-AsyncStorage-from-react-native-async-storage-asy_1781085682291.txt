const WISDOM_BOOKMARKS_KEY = "@girivalam/v2/wisdom-bookmarks";

async function getWisdomBookmarks(): Promise<string[]> {
  try {
    const raw = await AsyncStorage.getItem(WISDOM_BOOKMARKS_KEY);
    return raw ? (JSON.parse(raw) as string[]) : [];
  } catch {
    return [];
  }
}

async function toggleWisdomBookmark(id: string): Promise<string[]> {
  const list = await getWisdomBookmarks();
  const next = list.includes(id) ? list.filter((x) => x !== id) : [id, ...list];
  try {
    await AsyncStorage.setItem(WISDOM_BOOKMARKS_KEY, JSON.stringify(next));
  } catch {
    // Non-fatal; reading should not break if local storage fails.
  }
  return next;
}

function dayIndex(): number {
  const now = new Date();
  const start = new Date(now.getFullYear(), 0, 0);
  return Math.floor((now.getTime() - start.getTime()) / 86_400_000);
}

function pickDaily<T>(arr: T[]): T {
  return arr[dayIndex() % arr.length];
}

function timeAgo(ts: number): string {
  const seconds = Math.floor((Date.now() - ts) / 1000);
  if (seconds < 60) return "just now";
  const minutes = Math.floor(seconds / 60);
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  if (hours < 24) return `${hours}h ago`;
  const days = Math.floor(hours / 24);
  return `${days}d ago`;
}

function GuidanceModal({
  guide,
  visible,
  onClose,
  onOpenResource,
  onSaveInsight,
}: {
  guide: LifeGuide | null;
  visible: boolean;
  onClose: () => void;
  onOpenResource: (item: LibItem) => void;
  onSaveInsight: (payload: { situation: string; prompt: string; body: string }) => Promise<void>;
}) {
  const [selectedPrompt, setSelectedPrompt] = useState<string>("");
  const [insight, setInsight] = useState("");
  const [saving, setSaving] = useState(false);
  const [voiceHint, setVoiceHint] = useState(false);
  const [savedNotice, setSavedNotice] = useState(false);

  useEffect(() => {
    if (guide) {
      setSelectedPrompt(guide.reflections[0] ?? "");
      setInsight("");
      setVoiceHint(false);
      setSavedNotice(false);
    }
  }, [guide]);

  const resources = useMemo(
    () => (guide ? guide.resourceIds.map((id) => ITEM_BY_ID.get(id)).filter((x): x is LibItem => !!x) : []),
    [guide],
  );

  const save = async () => {
    if (!guide || !insight.trim() || saving) return;
    setSaving(true);
    try {
      await onSaveInsight({ situation: guide.title, prompt: selectedPrompt, body: insight.trim() });
      setInsight("");
      setSavedNotice(true);
    } finally {
      setSaving(false);
    }
  };

  return (
    <Modal visible={visible} transparent animationType="slide" onRequestClose={onClose}>
      <KeyboardAvoidingView
        behavior={Platform.OS === "ios" ? "padding" : undefined}
        style={styles.modalRoot}
      >
        <Pressable style={styles.modalBackdrop} onPress={onClose} />
        <View style={styles.guidanceSheet}>
          {guide ? (
            <>
              <View style={styles.sheetHandle} />
              <View style={styles.guidanceHeader}>
                <View style={styles.guidanceIconLarge}>
                  <Ionicons name={guide.icon} size={24} color={W.gold} />
                </View>
                <View style={{ flex: 1 }}>
                  <Text style={styles.guidanceOverline}>GUIDED INQUIRY</Text>
                  <Text style={styles.guidanceTitle}>{guide.title}</Text>
                  <Text style={styles.guidanceSubtitle}>{guide.doorway}</Text>
                </View>
                <Pressable onPress={onClose} hitSlop={10} accessibilityRole="button">
                  <Ionicons name="close" size={22} color={W.textMid} />
                </Pressable>
              </View>

              <ScrollView showsVerticalScrollIndicator={false} contentContainerStyle={styles.guidanceScroll}>
                <View style={styles.guidanceSection}>
                  <View style={styles.guidanceSectionTitleRow}>
                    <Ionicons name="sparkles-outline" size={15} color={W.gold2} />
                    <Text style={styles.guidanceSectionTitle}>Understanding</Text>
                  </View>
                  <Text style={styles.guidanceParagraph}>{guide.understanding}</Text>
                  <View style={styles.teachingQuoteWrap}>
                    {guide.teachings.map((t, index) => (
                      <View key={`${guide.key}-teaching-${index}`} style={styles.teachingQuote}>
                        <Text style={styles.teachingQuoteText}>“{t.text}”</Text>
                        <Text style={styles.teachingQuoteSource}>— {t.source}</Text>
                      </View>
                    ))}
                  </View>
                </View>

                <View style={styles.guidanceSection}>
                  <View style={styles.guidanceSectionTitleRow}>
                    <Ionicons name="help-circle-outline" size={15} color={W.gold2} />
                    <Text style={styles.guidanceSectionTitle}>Reflection</Text>
                  </View>
                  <Text style={styles.guidanceMuted}>Choose one question. Do not rush to answer it.</Text>
                  <View style={styles.promptList}>
                    {guide.reflections.map((prompt) => {
                      const active = selectedPrompt === prompt;
                      return (
                        <Pressable
                          key={prompt}
                          onPress={() => setSelectedPrompt(prompt)}
                          style={[styles.promptChip, active && styles.promptChipActive]}
                          accessibilityRole="button"
                        >
                          <Text style={[styles.promptChipText, active && styles.promptChipTextActive]}>{prompt}</Text>
                        </Pressable>
                      );
                    })}
                  </View>
                </View>

                <View style={styles.practiceBox}>
                  <View style={styles.practiceHeader}>
                    <View style={styles.practiceIconWrap}>
                      <Ionicons name="leaf-outline" size={16} color={W.gold} />
                    </View>
                    <View style={{ flex: 1 }}>
                      <Text style={styles.practiceTitle}>{guide.practice.title}</Text>
                      <Text style={styles.practiceDuration}>{guide.practice.duration}</Text>
                    </View>
                  </View>
                  <View style={styles.practiceSteps}>
                    {guide.practice.steps.map((step, index) => (
                      <View key={`${guide.key}-step-${index}`} style={styles.practiceStepRow}>
                        <Text style={styles.practiceStepNumber}>{index + 1}</Text>
                        <Text style={styles.practiceStepText}>{step}</Text>
                      </View>
                    ))}
                  </View>
                </View>

                <View style={styles.guidanceSection}>
                  <View style={styles.guidanceSectionTitleRow}>
                    <Ionicons name="library-outline" size={15} color={W.gold2} />
                    <Text style={styles.guidanceSectionTitle}>Wisdom resources</Text>
                  </View>
                  <Text style={styles.guidanceMuted}>Curated for this situation. Open only what feels useful now.</Text>
                  <View style={styles.resourceList}>
                    {resources.map((item) => (
                      <ResourceCard key={item.id} item={item} onOpen={onOpenResource} />
                    ))}
                  </View>
                </View>

                <View style={styles.insightBox}>
                  <View style={styles.guidanceSectionTitleRow}>
                    <Ionicons name="create-outline" size={15} color={W.gold2} />
                    <Text style={styles.guidanceSectionTitle}>Personal insight</Text>
                  </View>
                  <Text style={styles.insightPrompt}>{selectedPrompt}</Text>
                  <TextInput
                    value={insight}
                    onChangeText={setInsight}
                    multiline
                    placeholder="Write privately. A feeling, a sentence, a realization…"
                    placeholderTextColor={W.textFaint}
                    style={styles.insightInput}
                    textAlignVertical="top"
                  />
                  <View style={styles.insightActions}>
                    <Pressable
                      onPress={() => setVoiceHint((v) => !v)}
                      style={styles.voiceBtn}
                      accessibilityRole="button"
                    >
                      <Ionicons name="mic-outline" size={15} color={W.gold} />
                      <Text style={styles.voiceBtnText}>Voice reflection</Text>
                    </Pressable>
                    <Pressable
                      onPress={save}
                      disabled={!insight.trim() || saving}
                      style={[styles.saveInsightBtn, (!insight.trim() || saving) && { opacity: 0.45 }]}
                      accessibilityRole="button"
                    >
                      <Text style={styles.saveInsightText}>{saving ? "Saving…" : "Save insight"}</Text>
                    </Pressable>
                  </View>
                  {savedNotice ? (
                    <Text style={styles.savedHint}>Saved privately to Inner Reflection Space.</Text>
                  ) : null}
                  {voiceHint ? (
                    <Text style={styles.voiceHint}>
                      Voice notes need your app’s audio recording module. This front-end is ready for a private voice reflection flow.
                    </Text>
                  ) : null}
                </View>

                <Text style={styles.guidanceClosing}>{guide.closing}</Text>
              </ScrollView>
            </>
          ) : null}
        </View>
      </KeyboardAvoidingView>
    </Modal>
  );
}

function AIWisdomGuide({ onOpenResource, onSelectSituation }: { onOpenResource: (item: LibItem) => void; onSelectSituation: (guide: LifeGuide) => void }) {
  const [input, setInput] = useState("");
  const [reply, setReply] = useState<{ title: string; body: string; guide?: LifeGuide; resources: LibItem[] } | null>(null);

  const buildReply = useCallback((text: string) => {
    const lower = text.toLowerCase();
    const keywordKey =
      lower.includes("relationship") || lower.includes("partner") || lower.includes("family") ? "relationships" :
      lower.includes("loss") || lower.includes("lost someone") || lower.includes("grieving") ? "loss" :
      lower.includes("lost") || lower.includes("direction") || lower.includes("meaning") ? "purpose" :
      lower.includes("think") || lower.includes("loop") || lower.includes("worry") ? "overthinking" :
      lower.includes("uncertain") || lower.includes("future") || lower.includes("unknown") ? "uncertainty" :
      lower.includes("doubt") || lower.includes("enough") ? "self-doubt" :
      null;
    const matched = LIFE_GUIDES.find((g) =>
      lower.includes(g.key) ||
      lower.includes(g.title.toLowerCase()) ||
      g.title.toLowerCase().split("-").some((part) => part && lower.includes(part)),
    );
    const keywordMatched = keywordKey ? LIFE_GUIDES.find((g) => g.key === keywordKey) : null;
    const guide = matched ?? keywordMatched ?? LIFE_GUIDES[0];
    const resources = guide.resourceIds.map((id) => ITEM_BY_ID.get(id)).filter((x): x is LibItem => !!x).slice(0, 2);
    setReply({
      title: `A gentle place to begin: ${guide.title}`,
      body:
        `I cannot know your whole life from one sentence, but this sounds close to ${guide.title.toLowerCase()}. Begin by asking: “${guide.reflections[0]}” Then try the practice “${guide.practice.title}” for ${guide.practice.duration}.`,
      guide,
      resources,
    });
  }, []);

  const quickPrompts = ["I am overthinking", "I feel anxious", "I feel lost", "Relationship hurt"];

  return (
    <View>
      <SectionHeader
        overline="AI WISDOM GUIDE"
        title="Ask for a gentle direction"
        sub="A companion for suggestions, not spiritual authority. It points you back to personal inquiry."
      />
      <View style={styles.aiCard}>
        <View style={styles.aiHeaderRow}>
          <View style={styles.aiIconWrap}>
            <Ionicons name="sparkles-outline" size={18} color={W.gold} />
          </View>
          <View style={{ flex: 1 }}>
            <Text style={styles.aiTitle}>Wisdom assistant</Text>
            <Text style={styles.aiSub}>Describe what you are facing. Receive a starting point.</Text>
          </View>
        </View>
        <TextInput
          value={input}
          onChangeText={setInput}
          placeholder="Example: I keep worrying about the future…"
          placeholderTextColor={W.textFaint}
          style={styles.aiInput}
          multiline
          textAlignVertical="top"
        />
        <View style={styles.quickPromptRow}>
          {quickPrompts.map((p) => (
            <Pressable
              key={p}
              onPress={() => {
                setInput(p);
                buildReply(p);
              }}
              style={styles.quickPrompt}
              accessibilityRole="button"
            >
              <Text style={styles.quickPromptText}>{p}</Text>
            </Pressable>
          ))}
        </View>
        <Pressable
          onPress={() => buildReply(input || "anxiety")}
          style={styles.askGuideBtn}
          accessibilityRole="button"
        >
          <Text style={styles.askGuideText}>Suggest a path</Text>
          <Ionicons name="arrow-forward" size={15} color={W.whiteWarm} />
        </Pressable>

        {reply ? (
          <View style={styles.aiReply}>
            <Text style={styles.aiReplyTitle}>{reply.title}</Text>
            <Text style={styles.aiReplyBody}>{reply.body}</Text>
            <View style={styles.aiReplyActions}>
              {reply.guide ? (
                <Pressable onPress={() => onSelectSituation(reply.guide!)} style={styles.aiSmallBtn} accessibilityRole="button">
                  <Text style={styles.aiSmallBtnText}>Open guidance</Text>
                </Pressable>
              ) : null}
              {reply.resources.map((item) => (
                <Pressable key={item.id} onPress={() => onOpenResource(item)} style={styles.aiResourceBtn} accessibilityRole="button">
                  <Ionicons name={iconForKind(item.category)} size={13} color={W.gold} />
                  <Text style={styles.aiResourceText} numberOfLines={1}>{item.title}</Text>
                </Pressable>
              ))}
            </View>
          </View>
        ) : null}
      </View>
    </View>
  );
}

function StudyHallSection({
  query,
  setQuery,
  activeCategory,
  onSelectCategory,
  filtered,
  bookmarks,
  downloads,
  onOpen,
  onToggleBookmark,
  onToggleDownload,
  onRemoveDownload,
  onClearFilters,
}: {
  query: string;
  setQuery: (value: string) => void;
  activeCategory: LibraryKind | null;
  onSelectCategory: (kind: LibraryKind) => void;
  filtered: LibItem[];
  bookmarks: Set<string>;
  downloads: DownloadItem[];
  onOpen: (item: LibItem) => void;
  onToggleBookmark: (item: LibItem) => void;
  onToggleDownload: (item: LibItem) => void;
  onRemoveDownload: (id: string) => void;
  onClearFilters: () => void;
}) {
  const [openFull, setOpenFull] = useState(false);
  const downloadSet = useMemo(() => new Set(downloads.map((d) => d.id)), [downloads]);
  const filterActive = !!activeCategory || query.trim().length > 0;
  const featured = FEATURED_LIBRARY_IDS.map((id) => ITEM_BY_ID.get(id)).filter((x): x is LibItem => !!x);
  const display = filterActive ? filtered : openFull ? LIBRARY_ITEMS : featured;

  const toggleFull = () => {
    animateNext();
    setOpenFull((v) => !v);
  };

  return (
    <View>
      <SectionHeader
        overline="SACRED LIBRARY"
        title="Enter the study hall when you need depth"
        sub="Search, bookmark, download, and continue — but let the library support contemplation, not endless consumption."
      />
      <View style={styles.studyHallCard}>
        <View style={styles.studyHallIntro}>
          <MaterialCommunityIcons name="bookshelf" size={24} color={W.gold} />
          <View style={{ flex: 1 }}>
            <Text style={styles.studyHallTitle}>Sacred study hall</Text>
            <Text style={styles.studyHallSub}>Books, teachings, stories, chanting, audio, articles, and videos.</Text>
          </View>
        </View>

        <View style={styles.searchWrap}>
          <Ionicons name="search" size={17} color={W.textFaint} />
          <TextInput
            value={query}
            onChangeText={setQuery}
            placeholder="Search wisdom…"
            placeholderTextColor={W.textFaint}
            style={styles.searchInput}
            returnKeyType="search"
          />
          {query ? (
            <Pressable onPress={() => setQuery("")} hitSlop={8} accessibilityRole="button">
              <Ionicons name="close-circle" size={18} color={W.textFaint} />
            </Pressable>
          ) : null}
        </View>

        <CategoryNav active={activeCategory} onSelect={onSelectCategory} />

        {filterActive ? (
          <View style={styles.filterRow}>
            <Text style={styles.filterText} numberOfLines={1}>
              {activeCategory ? labelForKind(activeCategory) : "Search results"}
              {query.trim() ? ` · “${query.trim()}”` : ""}
            </Text>
            <Pressable onPress={onClearFilters} hitSlop={8} accessibilityRole="button">
              <Text style={styles.clearLink}>Clear</Text>
            </Pressable>
          </View>
        ) : null}

        <View style={styles.libraryList}>
          {display.length ? (
            display.map((item) => (
              <LibraryItemCard
                key={item.id}
                item={item}
                bookmarked={bookmarks.has(item.id)}
                downloaded={downloadSet.has(item.id)}
                onOpen={onOpen}
                onToggleBookmark={onToggleBookmark}
                onToggleDownload={onToggleDownload}
              />
            ))
          ) : (
            <View style={styles.emptyLibraryBox}>
              <Text style={styles.emptyText}>Nothing matches yet. Try another word or category.</Text>
            </View>
          )}
        </View>

        {!filterActive ? (
          <Pressable onPress={toggleFull} style={styles.openStudyBtn} accessibilityRole="button">
            <Text style={styles.openStudyText}>{openFull ? "Return to curated shelf" : "Open full study hall"}</Text>
            <Ionicons name={openFull ? "chevron-up" : "chevron-down"} size={15} color={W.gold} />
          </Pressable>
        ) : null}

        <View style={styles.downloadsPanel}>
          <View style={styles.downloadsHeader}>
            <Ionicons name="download-outline" size={16} color={W.gold2} />
            <Text style={styles.downloadsTitle}>Offline downloads</Text>
          </View>
          {downloads.length ? (
            <View style={{ gap: 2 }}>
              {downloads.slice(0, 6).map((d) => (
                <DownloadRow key={d.id} d={d} onOpen={(id) => {
                  const item = ITEM_BY_ID.get(id);
                  if (item) onOpen(item);
                }} onRemove={onRemoveDownload} />
              ))}
            </View>
          ) : (
            <Text style={styles.emptyText}>Downloaded readings will appear here for offline access.</Text>
          )}
        </View>
      </View>
    </View>
  );
}

function ReflectionSpace({ refreshKey = 0 }: { refreshKey?: number }) {
  const [notes, setNotes] = useState<InnerNote[]>([]);
  const [prompt, setPrompt] = useState(REFLECTION_PROMPTS[0]);
  const [draft, setDraft] = useState("");
  const [saving, setSaving] = useState(false);
  const [voiceHint, setVoiceHint] = useState(false);
  const mountedRef = React.useRef(true);

  const refresh = useCallback(async () => {
    const list = await getInnerNotes();
    if (mountedRef.current) setNotes(list);
  }, []);

  useEffect(() => {
    mountedRef.current = true;
    refresh();
    return () => {
      mountedRef.current = false;
    };
  }, [refresh, refreshKey]);

  const save = async () => {
    const body = draft.trim();
    if (!body || saving) return;
    setSaving(true);
    try {
      await addInnerNote({ body: `${prompt}\n\n${body}` });
      setDraft("");
      await refresh();
    } finally {
      setSaving(false);
    }
  };

  const remove = async (id: string) => {
    await removeInnerNote(id);
    animateNext();
    await refresh();
  };

  return (
    <View>
      <SectionHeader
        overline="INNER REFLECTION SPACE"
        title="Wisdom becomes yours when you reflect"
        sub="Private journal entries, insights, questions, gratitude, and voice reflection flow."
      />
      <View style={styles.reflectionCard}>
        <Text style={styles.reflectionIntro}>Choose a prompt or write freely. These reflections remain private on your device.</Text>
        <ScrollView horizontal showsHorizontalScrollIndicator={false} contentContainerStyle={styles.reflectionPromptScroll}>
          {REFLECTION_PROMPTS.map((p) => {
            const active = p === prompt;
            return (
              <Pressable
                key={p}
                onPress={() => setPrompt(p)}
                style={[styles.reflectionPromptChip, active && styles.reflectionPromptChipActive]}
                accessibilityRole="button"
              >
                <Text style={[styles.reflectionPromptText, active && styles.reflectionPromptTextActive]}>{p}</Text>
              </Pressable>
            );
          })}
        </ScrollView>
        <View style={styles.selectedPromptBox}>
          <Text style={styles.selectedPromptLabel}>TODAY’S PRIVATE QUESTION</Text>
          <Text style={styles.selectedPromptText}>{prompt}</Text>
        </View>
        <TextInput
          value={draft}
          onChangeText={setDraft}
          placeholder="Write what is true, not what sounds spiritual…"
          placeholderTextColor={W.textFaint}
          style={styles.noteInput}
          multiline
          textAlignVertical="top"
        />
        <View style={styles.reflectionActions}>
          <Pressable onPress={() => setVoiceHint((v) => !v)} style={styles.voiceBtn} accessibilityRole="button">
            <Ionicons name="mic-outline" size={15} color={W.gold} />
            <Text style={styles.voiceBtnText}>Voice note</Text>
          </Pressable>
          <Pressable
            onPress={save}
            disabled={!draft.trim() || saving}
            style={[styles.saveNoteBtn, (!draft.trim() || saving) && { opacity: 0.45 }]}
            accessibilityRole="button"
          >
            <Text style={styles.saveNoteText}>{saving ? "Saving…" : "Save reflection"}</Text>
          </Pressable>
        </View>
        {voiceHint ? (
          <Text style={styles.voiceHint}>
            Add Expo audio recording here to save private voice reflections. The navigation and UI pattern are prepared.
          </Text>
        ) : null}

        <Divider />
        <Text style={styles.subHead}>SAVED INSIGHTS</Text>
        {notes.length ? (
          <View style={styles.notesList}>
            {notes.slice(0, 8).map((note) => (
              <View key={note.id} style={styles.noteRow}>
                <Text style={styles.noteBody}>{note.body}</Text>
                <View style={styles.noteFooter}>
                  <Text style={styles.noteDate}>
                    {new Date(note.createdAt).toLocaleDateString(undefined, { month: "short", day: "numeric" })}
                  </Text>
                  <Pressable onPress={() => remove(note.id)} hitSlop={8} accessibilityRole="button">
                    <Ionicons name="trash-outline" size={15} color={W.textFaint} />
                  </Pressable>
                </View>
              </View>
            ))}
          </View>
        ) : (
          <Text style={styles.emptyText}>No reflections yet. One honest sentence is enough to begin.</Text>
        )}
      </View>
    </View>
  );
}

export default function WisdomScreen() {
  const insets = useSafeAreaInsets();
  const bottomInset = Platform.OS === "web" ? 34 : insets.bottom;
  const daily = useMemo(() => pickDaily(DAILY_WISDOM), []);

  const [selectedGuide, setSelectedGuide] = useState<LifeGuide | null>(null);
  const [readerItem, setReaderItem] = useState<LibItem | null>(null);
  const [query, setQuery] = useState("");
  const [activeCategory, setActiveCategory] = useState<LibraryKind | null>(null);
  const [continueRead, setContinueRead] = useState<LibraryProgress[]>([]);
  const [continueListen, setContinueListen] = useState<LibraryProgress[]>([]);
  const [recents, setRecents] = useState<RecentItem[]>([]);
  const [downloads, setDownloads] = useState<DownloadItem[]>([]);
  const [bookmarks, setBookmarks] = useState<string[]>([]);
  const [reflectionRefreshKey, setReflectionRefreshKey] = useState(0);

  const reload = useCallback(async () => {
    const [cr, cl, recent, dl, bm] = await Promise.all([
      getContinue("read"),
      getContinue("listen"),
      getRecentlyOpened(),
      getDownloads(),
      getWisdomBookmarks(),
    ]);
    setContinueRead(cr);
    setContinueListen(cl);
    setRecents(recent);
    setDownloads(dl);
    setBookmarks(bm);
  }, []);

  useEffect(() => {
    reload();
  }, [reload]);

  const bookmarkSet = useMemo(() => new Set(bookmarks), [bookmarks]);
  const downloadSet = useMemo(() => new Set(downloads.map((d) => d.id)), [downloads]);

  const openItem = useCallback(
    async (item: LibItem) => {
      setReaderItem(item);
      try {
        await recordOpened({ id: item.id, title: item.title, kind: item.category });
        const inProgress = item.category === ("book" as LibraryKind) || item.category === ("audiobook" as LibraryKind) || item.mode === "listen" || item.mode === "watch";
        await upsertLibraryProgress({
          id: item.id,
          title: item.title,
          kind: item.category,
          mode: item.mode === "watch" ? "read" : item.mode,
          progress: inProgress ? 0.22 : 1,
          position: item.mode === "listen" ? "Started listening" : item.mode === "watch" ? "Started watching" : inProgress ? "Started reading" : "Read",
        });
      } finally {
        reload();
      }
    },
    [reload],
  );

  const openById = useCallback(
    (id: string) => {
      const item = ITEM_BY_ID.get(id);
      if (item) openItem(item);
    },
    [openItem],
  );

  const onToggleBookmark = useCallback(async (item: LibItem) => {
    const next = await toggleWisdomBookmark(item.id);
    setBookmarks(next);
  }, []);

  const onToggleDownload = useCallback(
    async (item: LibItem) => {
      if (downloadSet.has(item.id)) {
        await removeDownload(item.id);
      } else {
        await addDownload({ id: item.id, title: item.title, kind: item.category, sizeLabel: item.audio ? "Audio" : item.video ? "Video" : "Text" });
      }
      reload();
    },
    [downloadSet, reload],
  );

  const onRemoveDownload = useCallback(
    async (id: string) => {
      await removeDownload(id);
      reload();
    },
    [reload],
  );

  const onSelectCategory = useCallback((kind: LibraryKind) => {
    setActiveCategory((cur) => (cur === kind ? null : kind));
  }, []);

  const filtered = useMemo(() => {
    let list = LIBRARY_ITEMS;
    if (activeCategory) list = list.filter((item) => item.category === activeCategory);
    const q = query.trim().toLowerCase();
    if (q) {
      list = list.filter((item) =>
        item.title.toLowerCase().includes(q) ||
        item.body.toLowerCase().includes(q) ||
        (item.author?.toLowerCase().includes(q) ?? false) ||
        labelForKind(item.category).toLowerCase().includes(q),
      );
    }
    return list;
  }, [activeCategory, query]);

  const clearFilters = useCallback(() => {
    setQuery("");
    setActiveCategory(null);
  }, []);

  const saveGuidedInsight = useCallback(async ({ situation, prompt, body }: { situation: string; prompt: string; body: string }) => {
    await addInnerNote({ body: `${situation}\n${prompt}\n\n${body}` });
    setReflectionRefreshKey((value) => value + 1);
  }, []);

  return (
    <>
      <ScrollView
        style={styles.container}
        contentContainerStyle={[styles.content, { paddingBottom: bottomInset + 42 }]}
        showsVerticalScrollIndicator={false}
        keyboardShouldPersistTaps="handled"
      >
        <OpeningAshram daily={daily} />

        <LifePromptSection onSelect={setSelectedGuide} />

        <ContinueJourneySection
          continueRead={continueRead}
          continueListen={continueListen}
          recents={recents}
          onOpen={openById}
        />

        <AIWisdomGuide onOpenResource={openItem} onSelectSituation={setSelectedGuide} />

        <WalkingWithRamanaSection onOpenResource={openItem} />

        <StudyHallSection
          query={query}
          setQuery={setQuery}
          activeCategory={activeCategory}
          onSelectCategory={onSelectCategory}
          filtered={filtered}
          bookmarks={bookmarkSet}
          downloads={downloads}
          onOpen={openItem}
          onToggleBookmark={onToggleBookmark}
          onToggleDownload={onToggleDownload}
          onRemoveDownload={onRemoveDownload}
          onClearFilters={clearFilters}
        />

        <ReflectionSpace refreshKey={reflectionRefreshKey} />

        <AccessCard />

        <View style={styles.closingWrap}>
          <MaterialCommunityIcons name="meditation" size={22} color={W.gold} />
          <Text style={styles.closingText}>Leave with one small clarity. Return when life asks another question.</Text>
        </View>
      </ScrollView>

      <GuidanceModal
        visible={!!selectedGuide}
        guide={selectedGuide}
        onClose={() => setSelectedGuide(null)}
        onOpenResource={(item) => {
          setSelectedGuide(null);
          openItem(item);
        }}
        onSaveInsight={saveGuidedInsight}
      />

      <ReaderModal
        item={readerItem}
        bookmarked={readerItem ? bookmarkSet.has(readerItem.id) : false}
        downloaded={readerItem ? downloadSet.has(readerItem.id) : false}
        onClose={() => setReaderItem(null)}
        onToggleBookmark={onToggleBookmark}
        onToggleDownload={onToggleDownload}
      />
    </>
  );
}
